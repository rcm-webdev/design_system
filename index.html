GitHub Actions Workflow

name: Deploy to GitHub Pages
on:
  push:
    branches: [ main ]
jobs:
  deploy:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Compile Sass
        run: sass scss/taro.scss style.css
      - name: Deploy
        uses: peaceiris/actions-gh-pages@v4
